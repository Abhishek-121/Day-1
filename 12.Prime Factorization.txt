-->1. You are required to display the prime factorization of a number.
    2. Take as input a number n.
    3. Print all its prime factors from smallest to largest.

--ANS--
#include<iostream>
#include<bits/stdc++.h>
using namespace std;
int main()
{
    int n;
    cin>>n;
    for(int i=2;i<sqrt(n);i++)
    {
        while(n%i==0)
        {
            n=n/i;
            cout<<i<<" ";
        }
    }
    if(n!=1)
    {
        cout<<n<<" ";
    }
}
